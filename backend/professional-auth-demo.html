// Install the professional auth client import { authClient } from './lib/professional-auth' // Login function const login = async (email: string, password: string) => { const result = await authClient.login(email, password) if (result.success) { console.log('Logged in as:', result.user.email) return true } else { console.error('Login failed:', result.error) return false } } // Get current user const getCurrentUser = async () => { const result = await authClient.getCurrentUser() if (result.success) { return result.user } else { console.log('Not authenticated:', result.error) return null } } // Make authenticated API calls const apiCall = async (endpoint: string) => { try { const data = await authClient.apiCall(endpoint) return data } catch (error) { console.error('API call failed:', error) throw error } }
